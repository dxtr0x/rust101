use std::{fs::File, io::{Read, Write}};

fn main() {
    let mut txt = String::new();
    let mut file = File::create("foo.txt").unwrap();
    for _ in 1..10 {
        file.write(b"Hello world\n").unwrap();
    }

    let mut read_f = File::open("foo.txt").unwrap();
    read_f.read_to_string(&mut txt);

    print!("Text : {}",txt);
}
